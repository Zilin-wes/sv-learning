# 02数组
## 1.声明
```systemverilog
byte array_0 [8]; //8个独立的成员变量 byte后不能加[] 即byte [3:0] array_2 [0:7];是错的 因为byte有固定8bit无法修改
byte array_1[0:7][0:3]; //8x4个独立的

bit [3:0] array_2 [0:7][0:3]; // packed array unpacked 8x4个独立的成员变量,成员变量是4个bit
bit [3:0][1:0] array_3 [0:7][0:3]; //次高级 最低级 array 最高级 8x4个独立的成员变量,成员变量是4个packed的2个bit

logic [7:0] array_4 [0:3];  //四值变量默认x 二值默认0

int ids;

initial begin
//1.赋值
  array_0[0]= 8'h55;  //单个赋值:<位宽>'<进制><数值>  b二进制 h十六进制 o八进制 d十进制
  array_1[2]='{8'h00, 8'h01, 8'h02, 8'h03};  //成组赋值'{}
// unpacked 不能赋值给packed

//2.元素个数
  ids= $size(array_3);  //返回当前维度的元素个数8
  #10;
  ids= $size(array_3[1]);  //4
  #10;
  ids= $size(array_3[1][0]);  //4
  #10;
  ids= $size(array_3[1][0][0]);  //2
  #10;
  ids= $size(array_3[1][0][0][0]);  //0

//3.循环(类似for)  
  foreach(array_0[i]) begin
      $display("array_0[%0d]=%0h.", i, array_0[i]);
  //output: array_0[0]=85. 从小到大
  //array_0[1]=0.
  //array_0[2]=0. ...

  foreach(array_2[i, j]) begin
      $display("array_2[%0d][%0d]=%0h.", i,j, array_2[i][j]);  //foreach不能遍历packed
 //output: array_2[0][0]=0.
  //array_2[0][1]=0.
  //array_2[0][2]=0. ...
  
end
```

## 2.动态数组

```systemverilog

int dyn[], d2[];
byte d3[]='{-1, 0, 1}; //声明三个动态数组 

innitial begin
  dyn=new[5]; //操作前必须用new命令分配数组大小 此处分配5个
  foreach(dyn[i])
      dyn[i]=i*2;
  d2=dyn;
  d2[0]=6;
  $display(dyn[0], d2[0]);

  dyn=new[10](dyn);//分配10个cell 0-4copy原来的dyn
  foreach(dyn[i])
      $write("%0d", dyn[i]);

  dyn.delete(); //删除内存分配
end
```
